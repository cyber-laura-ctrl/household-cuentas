function addEntry(){

  setMonth(m=>({...m,variable:[...m.variable,e]}));

}

  if(isFixed) setMonth(m=>({...m,fixed:m.fixed.filter(e=>e.id!==id)}));

}

  state.editId=null;

}

// ── calculations ───────────────────────────────────────────────

function calcMonth(m){

  const all=[...m.fixed,...m.variable];

  const exp=all.filter(e=>e.type==="expense"), sav=all.filter(e=>e.type==="savings");

  const totExp=exp.reduce((s,e)=>s+Number(e.amount),0);

  const totSav=sav.reduce((s,e)=>s+Number(e.amount),0);

  const pAe=exp.filter(e=>e.paidBy===PA).reduce((s,e)=>s+Number(e.amount),0);

  const pBe=exp.filter(e=>e.paidBy===PB).reduce((s,e)=>s+Number(e.amount),0);

  const pAs=sav.filter(e=>e.paidBy===PA).reduce((s,e)=>s+Number(e.amount),0);

  const pBs=sav.filter(e=>e.paidBy===PB).reduce((s,e)=>s+Number(e.amount),0);

  const netA=pAe-totExp*SA;

  return {exp,sav,totExp,totSav,pAe,pBe,pAs,pBs,

    shareAe:totExp*SA,shareBe:totExp*SB,shareAs:totSav*SA,shareBs:totSav*SB,netA};

}

    return `<div class="hist-row">

      <div class="top">

        <span style="font-size:14px;font-weight:700;min-width:130px">${label}</span>

        <span style="font-size:13px;color:#94a3b8">Expenses: <b style="color:#f1f5f9">${fmt(c.totExp)}</b></span>

        <span style="font-size:13px;color:#94a3b8">Savings: <b style="color:#f59e0b">${fmt(c.totSav)}</b></span>

        ${status}

      </div>

      ${m.settlement?.note?`<div class="note">Note: ${m.settlement.note}</div>`:""}

    </div>`;

  }).join("");

  return `<div class="card"><div class="card-title">Monthly History</div>${rows}</div>`;

}

  if(!state.showModal) return "";

  const {netA}=c, owedAmt=Math.abs(netA), owedBy=netA>0?PB:PA, owedTo=netA>0?PA:PB;

  return `<div class="overlay">

    <div class="modal">

      <h2>Mark Settlement</h2>

      <p>${owedBy} pays ${owedTo} <span style="color:#f59e0b;font-weight:700">${fmt(owedAmt)}</span></p>

      <div class="field"><label>Date of payment</label><input type="date" value="${state.settleDate}" oninput="state.settleDate=this.value"></div>

      <div class="field"><label>Note (optional)</label><input placeholder="e.g. e-transfer, cash…" value="${state.settleNote}" oninput="state.settleNote=this.value"></div>

      <div class="modal-btns">

        <button style="background:#22c55e;color:#fff;border:none" onclick="markSettled('${owedBy}','${owedTo}',${owedAmt})">Confirm Settled</button>

        <button class="btn-ghost" onclick="state.showModal=false;render()">Cancel</button>

      </div>

    </div>

  </div>`;

}
